# Comandos para subir o projeto no GitHub
# Execute um por vez no terminal após instalar o Git

# 1. Configurar seu nome e email (substitua pelos seus dados)
git config --global user.name "Seu Nome"
git config --global user.email "seu-email@exemplo.com"

# 2. Inicializar o repositório
git init

# 3. Adicionar todos os arquivos
git add .

# 4. Fazer o primeiro commit
git commit -m "Primeiro commit - Sistema de seleção de matérias"

# 5. Adicionar o repositório remoto
git remote add origin https://github.com/88MVcode9/mvupoestudo.git

# 6. Configurar a branch principal
git branch -M main

# 7. Enviar para o GitHub (vai pedir usuário e senha/token)
git push -u origin main
